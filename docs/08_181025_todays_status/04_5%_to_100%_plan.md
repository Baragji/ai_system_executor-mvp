### 1A) `src/orchestrator/graph.ts` — multi-node StateGraph with durability & HITL

**Goal:** Build a real graph with these nodes and flow:

`START → clarify → plan → generate → test → [pass? deliver : repair → test(loop)] → END`

* *Clarify* (optional questions).
* *Plan* (build a task plan).
* *Generate* (use **real** LLM to write files).
* *Test* (run sandbox/tests).
* *Repair* (LLM-guided code changes if tests fail; loop with ceiling).
* *Deliver* (complete + persist results).

**Exact edits (create/replace file content):**

```ts
// FILE: src/orchestrator/graph.ts
import { Annotation, StateGraph, START, END } from "@langchain/langgraph";
import { MemorySaver } from "@langchain/langgraph-checkpoint";
// HITL (interrupts) & parallel send are available from runtime context in JS docs.
// We'll use interrupts in a node via the second arg.
import { updateExecution, completeExecution, failExecution, createExecution } from "./executionsStore";
import * as Clar from "../clarification/generateQuestions";
import * as Plan from "../planning/decomposeTask";
import * as Exec from "../planning/executeTaskPlan";
import * as Repair from "../repair/multiTurnRepair";
import * as Runner from "../runner/runInSandbox";
import type { ExecutionRecord } from "./executionTypes";

type TestResult = { status: "passed" | "failed"; details?: any };
type FilesSpec = Array<{ path: string; contents: string }>;

const S = Annotation.Root({
  prompt: Annotation<string>(),
  projectName: Annotation<string | undefined>(),
  plan: Annotation<any | undefined>(),
  files: Annotation<FilesSpec>({ reducer: (a = [], b = []) => [...a, ...b] }),
  test: Annotation<TestResult | undefined>(),
  repair: Annotation<{ attempts: number; lastNote?: string } | undefined>(),
  logs: Annotation<string[]>({ reducer: (a = [], b = []) => a.concat(b) }),
  status: Annotation<"running" | "completed" | "failed">(),
  error: Annotation<unknown | undefined>(),
});

const builder = new StateGraph(S);

// ---- nodes ----
builder.addNode("clarify", async (state, { interrupts }) => {
  const qs = await Clar.generateQuestions(state.prompt);
  // Pause for HITL approval of the plan inputs (can be auto-approved in dev)
  // When paused, resume via the server’s approve endpoint (added below).
  const approval = await interrupts?.interrupt?.("approve_clarifications", { questions: qs });
  return { logs: [`clarify: approved=${approval !== undefined}`] };
});

builder.addNode("plan", async (state) => {
  const plan = await Plan.decomposeTask({ prompt: state.prompt });
  return { plan, logs: ["plan: built"] };
});

builder.addNode("generate", async (state) => {
  const files = await Exec.executeTaskPlan({
    plan: state.plan!,
    projectName: state.projectName ?? "project",
    // ExecuteTaskPlan should route to the repo’s real LLM provider layer.
  });
  return { files, logs: [`generate: wrote ${files.length} file(s)`] };
});

builder.addNode("test", async (state) => {
  const r = await Runner.runInSandbox({ project: state.projectName ?? "project" });
  const status: TestResult["status"] = r?.status === "passed" ? "passed" : "failed";
  return { test: { status, details: r }, logs: [`test: ${status}`] };
});

builder.addNode("needs_repair", async (state) => {
  return state.test?.status === "passed" ? "deliver" : "repair";
});

builder.addNode("repair", async (state) => {
  const attempt = (state.repair?.attempts ?? 0) + 1;
  if (attempt > 3) {
    return { status: "failed", logs: ["repair: giving up after 3 attempts"] };
  }
  const out = await Repair.multiTurnRepair({
    prompt: state.prompt,
    testResult: state.test!,
    projectName: state.projectName ?? "project",
  });
  return {
    repair: { attempts: attempt, lastNote: out?.note ?? "" },
    files: out?.files ?? [],
    logs: [`repair: attempt ${attempt}`],
  };
});

builder.addNode("deliver", async (state) => {
  return { status: "completed", logs: ["deliver: done"] };
});

// ---- edges ----
builder.addEdge(START, "clarify");
builder.addEdge("clarify", "plan");
builder.addEdge("plan", "generate");
builder.addEdge("generate", "test");
builder.addConditionalEdges("test", "needs_repair", {
  deliver: "deliver",
  repair: "repair",
});
builder.addEdge("repair", "test");
builder.addEdge("deliver", END);

// ---- compile with a checkpointer so we get persistence/threads
// (MemorySaver is fine for dev; swap to persistent saver later)
const app = builder.compile({ checkpointer: new MemorySaver() });

export async function runWithLangGraph(input: {
  prompt: string;
  projectName?: string;
}): Promise<ExecutionRecord> {
  // We create an execution record BEFORE invoking the graph.
  const rec = await createExecution({ status: "started", metadata: { runtime: "langgraph" } });
  const executionId = rec.id;

  try {
    // Threaded persistence: pass thread_id so halts/resumes work and state is durable.
    const finalState = await app.invoke(
      { prompt: input.prompt, projectName: input.projectName, logs: ["graph: start"] },
      { configurable: { thread_id: executionId } }
    );

    if (finalState.status === "failed") {
      await failExecution(executionId, { error: finalState.error, logs: finalState.logs });
    } else {
      await completeExecution(executionId, { result: finalState, logs: finalState.logs });
    }
    return (await updateExecution(executionId, { status: finalState.status }))!;
  } catch (err) {
    await failExecution(executionId, { error: err, logs: ["graph: exception"] });
    throw err;
  }
}
```

**Why these choices (brief):**

* `StateGraph` + `addNode/addEdge/addConditionalEdges` implement the real DAG/loop.
* `compile({ checkpointer: new MemorySaver() })` gives **durable execution** with resumability by `thread_id`. For production, swap to a persistent saver; `MemorySaver` is fine for local validation. ([docs.langchain.com][1])
* `interrupt(...)` pauses a node for **human-in-the-loop** and resumes when you post a response (we’ll add a tiny approve endpoint). ([docs.langchain.com][2])

3. **(Optional) Approve endpoint for HITL pause**
   Add a tiny endpoint to resume `clarify` pauses. For dev we can auto-approve, but expose a hook:

```ts
app.post("/api/executions/:id/approve", async (req, res) => {
  // For MemorySaver, resuming is implicit when the graph is invoked again with same thread_id.
  // A minimal approach: store an "approved" flag; clarify node checks interrupts; in MemorySaver,
  // the first await interrupts.interrupt() will just return this payload if provided.
  // For now, we return 204 to indicate "approved"; clarify node uses .interrupt() to pause,
  // and the next invoke continues. In dev we auto-approve by not awaiting an external event.
  res.status(204).end();
});
```

> Note: LangGraph’s interrupt contract pauses execution until the user supplies input; with **threads + checkpointer**, you resume by sending the value back into the thread. The minimal “dev” path above treats interrupts as auto-approved; later you can wire a proper resume payload. ([docs.langchain.com][2])

```ts
const RUN = process.env.RUN_REAL_LLM === "1";

  const resp = await fetch(`http://localhost:${PORT}/api/execute`, {

      prompt: "Make a minimal Node+TS Hello World web app with GET / that returns 'Hello World'. Include README with run steps.",
    }),
  });
  expect(resp.status).toBe(202);
  const body: any = await resp.json();
  expect(body.executionId).toBeTruthy();

  // poll until completed
  const execUrl = `http://localhost:${PORT}/api/executions/${body.executionId}`;
  for (let i = 0; i < 120; i++) {
    await new Promise(r => setTimeout(r, 1000));
    const r = await fetch(execUrl);
    const j: any = await r.json();
    if (j.status === "completed") {
      expect(j.result).toBeTruthy();
      return;
    }
    if (j.status === "failed") throw new Error("Execution failed");
  }
  throw new Error("Timed out waiting for completion");
});
```
